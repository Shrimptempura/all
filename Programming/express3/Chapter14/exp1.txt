<더블 포인터 기초 중에 기초>

*p = 200;       // 변수 i의 값이 200으로 변경됨
**q = 300;      // 변수 i의 값이 300으로 변경됨

    *q는 p와 같게 되므로 밑에 문장은 논리적 오류이다.
    *q = 300;       
    포인터 p에 주소 300을 대입하는 것은 의미가 없다.
    우리가 의도한 바는 변수 i의 값을 300으로 변경하는 것이다.

// 이중 포인터 기초..
#include <stdio.h>

int main(void)
{
    int i = 100;
    int *p = &i;    // 포인터 p는 i를 가리킨다.
    int **q = &p;   // 이중 포인터 q는 p를 가리킨다.

    *p = 200;       // p를 통하여 i에 200저장
    printf("i = %d\n", i);

    **q = 300;      // q를 통하여 i에 300저장
    printf("i = %d\n", i);

    return 0;
}
